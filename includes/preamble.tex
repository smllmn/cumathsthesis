% here I define a chapter style for the document. it is based on the "dash" style
% provided by memoir. in fact, it first sets the chapterstyle to dash to import
% its settings, then changes what I didn't like (mostly font sizes)
\makeatletter
\makechapterstyle{fauxdash}{%
    \chapterstyle{dash}
    \setlength{\beforechapskip}{2\onelineskip}
    \renewcommand*{\chapnumfont}{\normalfont\LARGE}
    \renewcommand*{\printchapternum}{\centering \chapnumfont 
        \rule[0.5\midchapskip]{1em}{0.4pt} \@chapapp{} \thechapter\ 
        \rule[0.5\midchapskip]{1em}{0.4pt}
    }
    \renewcommand*{\chaptitlefont}{\normalfont\Huge}
}
\makeatother
